# git 初始化 在当前目录下
        git init

# git 仓库添加文件 该文件必须在git仓库目录下或者子目录下
        git add + 文件名

# 添加之后要提交进仓库中
        git commit -m "输入一些提示语"

# 修改了库中的文件之后，如何查看当前库的状态
        git status

# 查看库的状态只能看见文件被修改等等，并不能看到具体的修改部分
        git diff + 文件路径/文件名

# 查看最近修改的几个版本具体是谁，修改的提示词是什么
        git log 
# 如果嫌弃字太多
        git log --pretty=oneline
# 按q退出

# 如何退回版本,HEAD^ HEAD~100
        --hard 退回该版本已经上传的状态
        --soft 退回该版本未提交的状态
        --mixed 退回上个版本已添加但未提交的状态

        git reset --hard HEAD^

# 如果后悔了想要回到当时最新版本，需要知道该版本的版本号（在命令行中）
        git reset --hard 70e4f （可以多写几个让git去找）

# 如果关闭了电脑，命令找不到了怎么办,下面这个命令会打印你输入过的指令
        git reflog

# 管理修改 如果文件修改还未提交，回复到版本最新状态，
# 如果已经提交一次修改，回复到提交修改后的状态
        git checkout -- filename

# git有工作区、缓存区、版本区,工作区直接删除文件后
# 如果真的想删除
        git rm + 文件名
        并且，更新缓存区
        git commit             版本区也删除了
# 如果是误删
        git checkout -- 文件名 就能在工作区恢复文件，但是只能恢复到版本最新，修改会丢失


# 本地git仓库链接github
        先在本地创建.ssh文件,得到公钥和私钥之后，先在github上添加公钥之后在本地创建链接
        git remote add origin git@github.com:xiequan277/GET-GPT.git

        仓库分支推送
        git push -u origin master

# 远程仓库创建分支并切换
        git checkout -b dev = git branch dev + git checkout dev

# git 分支与分支之间的工作合并
        git merge 2025.8.4
